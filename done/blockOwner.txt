@startuml

actor "Admin" as user
participant "Gateway" as gw
participant "User App" as us
participant "Hotel App" as hs

database "UserDB" as ub
database "HotelDB" as hb

==Blocking the owner and hotel==
    activate user
user -> gw :PUT /owners?status=block
    activate gw
gw -> us: send request to block\n the owner and hotel
    activate us
us -> ub: update an owner to status block
    activate ub
ub -> us: return the owner
    deactivate ub
us -> hs: send request\n to block the hotel
    activate hs
hs -> hb: update the hotel to status block
    activate hb
hb -> hs: return the hotel
    deactivate hb
hs -> us: return the hotel
    deactivate hs
us -> gw: return the owner and hotel view
    deactivate us
gw -> user: return the owner and hotel view
    deactivate gw
    deactivate user

@enduml

@startuml

actor "Owner" as user
participant "Gateway" as gw
participant "Hotel App" as hs
database "HotelDB" as db

==Adding minimum requirements for the rent room to the room==
    activate user
user -> gw: POST /rooms/{id}/requirements
    activate gw
gw -> hs: send a request to add minimum\nrequirements for rent room
    activate hs
hs-> db: update a minimum requirements\n for rent room in status active
    activate db
db-> hs: return the room in status active
    deactivate db
hs -> gw: return the room view
    deactivate hs
gw ->  user: return the room view
    deactivate gw
    deactivate user


@enduml
